-- Habilitar extensión PostGIS si no está
create extension if not exists postgis;

-- Recrear tabla reportes (borra la existente si ya la tenías, cuidado con datos previos)
drop table if exists reportes cascade;

create table reportes (
  id bigint generated always as identity primary key,
  lat double precision not null,
  lng double precision not null,
  tipo_precursor text not null,
  peso integer not null,
  descripcion text default '',
  created_at timestamptz default now(),
  geom geometry(Point, 4326) generated always as (
    st_setsrid(st_makepoint(lng, lat), 4326)
  ) stored
);

-- Índice espacial
create index idx_reportes_geom on reportes using gist (geom);

-- Políticas de seguridad
alter table reportes enable row level security;
create policy "anon_insert" on reportes for insert to anon with check (true);
create policy "anon_select" on reportes for select to anon using (true);

-- Función para obtener reportes cercanos
create or replace function reportes_cercanos(
  p_lat double precision,
  p_lng double precision,
  radius_meters int default 1000
)
returns table (
  id bigint,
  tipo_precursor text,
  peso int,
  created_at timestamptz,
  distancia_m float
) as $$
begin
  return query
  select r.id, r.tipo_precursor, r.peso, r.created_at,
         st_distance(r.geom::geography, st_setsrid(st_makepoint(p_lng, p_lat), 4326)::geography) as distancia_m
  from reportes r
  where st_dwithin(r.geom::geography, st_setsrid(st_makepoint(p_lng, p_lat), 4326)::geography, radius_meters)
    and r.created_at > now() - interval '72 hours'
  order by distancia_m;
end;
$$ language plpgsql;
